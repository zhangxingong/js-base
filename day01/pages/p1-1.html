1-1函数作为参数

案例1-1：

见证奇迹的时刻..........

         function foo1() {
                alert("我是一个普通函数");
            }

            function foo2(fun){
                alert("我是一个特殊的参数，调用了函数"+fun);
                alert("调用"+fun+"开始");
                fun();
                alert("调用"+fun+"结束");
            }

            //调用一下 函数当做参数
            foo2(foo1);
