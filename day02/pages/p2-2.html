            function fruit(name,color,taste){
                this.name = name;
                this.color = color;
                this.taste = taste;
                this.info = function(){
                    alert("我是"+this.name+" ,我的颜色是"+this.color+" ,我的味道是"+this.taste);
                }
            }

            var apple = new fruit("苹果","红色","甜味");
