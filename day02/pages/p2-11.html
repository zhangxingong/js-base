                //实现一个类对一个雷的继承
                function Person() {
                    this.type = "人";
                }
                Person.prototype.getType = function(){
                    alert("这是一个人");
                }

                function Student(_name,_age){
                    Person.call(this);
                    this.name  = _name;
                    this.age = _age;
                }
                Student.prototype.say = function(){
                    alert("我是一个学生");
