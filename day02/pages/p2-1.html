            //构造水果对象的工厂函数
            function createFruit(name,color,taste){
                var fruit = {
                    name:name,
                    color:color,
                    taste:taste
                }
                return fruit;
            }

            var apple = createFruit("苹果","红色","甜味");
            var banan = createFruit("香蕉","黄色","香味");
