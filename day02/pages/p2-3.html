           //创建水果类的构造函数
            function fruit(name,color,taste){
                this.name = name;
                this.color = color;
                this.taste = taste;
                this.info = function(){
                    alert("我是"+this.name+" ,我的颜色是"+this.color+" ,我的味道是"+this.taste);
                }
            }

            var apple = new fruit("苹果","红色","甜味");
            var banana = new fruit("香蕉","黄色","香味");

            alert(apple instanceof fruit);
            alert(banana instanceof fruit);
            alert(apple.hasOwnProperty("name"));
            alert(apple.hasOwnProperty("color"));
            alert(apple.hasOwnProperty("taste"));
            alert("name"   in apple);
            alert("color"  in apple);
            alert("taste"  in apple);
