            //创建水果类的
            function fruit(name,color,taste){
                constructor : fruit,
                this.name = name;
                this.color = color;
                this.taste = taste;
                if(typeof  this.info != "function"){
                    fruit.prototype = {
                        info  : function(){
                            alert("我是"+this.name+" ,我的颜色是"+this.color+" ,我的味道是"+this.taste);
                        }
                    }
                }
            }


            var apple = new fruit();
            var banana = new fruit();

            alert(apple.constructor);
            alert(apple.info());
            alert(banana.info());
