            //创建水果类的
            function fruit(){}
            fruit.prototype.name = "苹果";
            fruit.prototype.color = "红色";
            fruit.prototype.taste = "甜味";
            fruit.prototype.info = function(){
                alert("我是"+this.name+" ,我的颜色是"+this.color+" ,我的味道是"+this.taste);
            }

            var apple = new fruit();
            var banana = new fruit();
            alert(apple === banana);
            alert(apple.isPrototypeOf("fruit"));
            alert(banana.isPrototypeOf("fruit"));
