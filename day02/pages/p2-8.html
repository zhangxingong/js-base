            //使用对象冒充的方法实现继承
            function Monkey(type,name){
                this.type = type;
                this.name = name;
            }
            function Hero(_HP){
                this.HP = _HP;
            }

            function Magic_Monkey(type,name,skill,_HP){
                Monkey.apply(this,[type,name]);
                Hero.call(this,_HP);
                this.skill = skill;
            }

            var wukong = new Monkey("猴子","孙悟空");

            var dasheng = new Magic_Monkey("三十二变");

            alert(wukong.name);
            alert(wukong.type);
            alert(wukong.skill);
            alert(dasheng.type);
            alert(dasheng.name);
