  • 匿名函数传递参数

            simplefn();
            var anoymity =  function(){

            //-----------------END--------------

            //-----------------START-----------

            (function(){
                alert("我是匿名函数，我立即执行了。")
            })()
            //-----------------END----------------
            //传递参数
            //-----------------START--------------
            (function(value1,value2){
                alert(value1 + value2);
