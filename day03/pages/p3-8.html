                function people(_name,_age){
                    var name = _name; //私有变量
                    var age  = _age;  //私有变量
                    var say  = function(){alert("我是一个方法")} //私有方法
                    //特权方法
                    this. _getName = function(){
                        return name;
                    }
                    this._setName = function(_name_){
                        name = _name_;
                    }
                    this._getAge  = function(){
                        return age;
                    }
                    this._setAge = function(_age_){
                        age = _age_;
                    }
                    this._getSay  = function(){
                        return say();
                    }
                }
                var zhangsn = new people("张三",20);
                alert(zhangsn._getName());
                alert(zhangsn._getAge());
                alert(zhangsn._getSay());
                zhangsn._setName("李四")
                alert(zhangsn._getName());
            // -----------END------------------------
